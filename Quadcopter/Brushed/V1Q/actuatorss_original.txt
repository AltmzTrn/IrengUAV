#include "actuators.h"
#include "controlSystems.h"

void writeServoMicros(uint8_t pin, uint16_t us) {
  // On STM32 Arduino core, analogWrite works in range 0–255 by default (8-bit)
  // But we want to map microseconds (e.g. 1000–2000us) into that
  // Assuming 50Hz (20ms period), 1000us = 5% duty cycle, 2000us = 10% duty

  // STM32 analogWrite() resolution is set using analogWriteResolution()
  // Let’s use full 16-bit resolution
  analogWriteResolution(16);  // 0 - 65535

  const uint32_t pwmFreq = 50;           // 50 Hz for servos and ESC
  const uint32_t timerPeriod = 1000000 / pwmFreq; // Period in microseconds (20ms)

  // Convert desired pulse width to duty
  uint32_t duty = (us * 65535UL) / timerPeriod;
  analogWrite(pin, duty);
}

void actuators_setup() {
  pinMode(motor_R_F, OUTPUT);
  pinMode(motor_L_F, OUTPUT);
  pinMode(motor_R_R, OUTPUT);
  pinMode(motor_L_R, OUTPUT);

  analogWriteFrequency(50);
}


void actuators_write() {
  // Write directly as microsecond PWM signals
  writeServoMicros(motor_R_F, to_actuator[0]);
  writeServoMicros(motor_L_F, to_actuator[1]);
  writeServoMicros(motor_R_R, to_actuator[2]);
  writeServoMicros(motor_L_R, to_actuator[3]);
}
