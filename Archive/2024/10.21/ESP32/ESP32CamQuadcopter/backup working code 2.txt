// Function to handle joystick 1 data
void handleJoystick1() {
  if (server.hasArg("x") && server.hasArg("y")) {
    x = server.arg("x").toInt();
    y = server.arg("y").toInt();
    Serial.printf("Joystick 1 - X: %d Y: %d\n", x, y);
  }
  server.send(200, "text/plain", "Joystick 1 values received");
}

// Function to handle joystick 2 data
void handleJoystick2() {
  if (server.hasArg("x2") && server.hasArg("y2")) {
    x2 = server.arg("x2").toInt();
    y2 = server.arg("y2").toInt();
    Serial.printf("Joystick 2 - X2: %d Y2: %d\n", x2, y2);
  }
  server.send(200, "text/plain", "Joystick 2 values received");
}

void handleStream() {
  camera_fb_t* fb = esp_camera_fb_get();
  if (!fb) {
    Serial.println("Camera capture failed");
    return;
  }

  // Set headers for the response
  server.sendHeader("Access-Control-Allow-Origin", "*");
  server.sendHeader("Cache-Control", "no-cache");
  
  // Create a temporary file to hold the image data
  WiFiClient client = server.client(); // Get the client instance
  if (client) {
    client.write(fb->buf, fb->len); // Send the image buffer directly
  }
  
  // Return the frame buffer back to the driver for reuse
  esp_camera_fb_return(fb);
}

void setup() {
  Serial.begin(115200);
  // Connect to Wi-Fi
  WiFi.begin(ssid, password);
  while (WiFi.status() != WL_CONNECTED) {
    delay(500);
    Serial.print(".");
  }
  Serial.println("Connected to WiFi");

  // Start camera
  camera_config_t config;
  config.ledc_channel = LEDC_CHANNEL_0;
  config.ledc_timer = LEDC_TIMER_0;
  config.pin_d0 = Y2_GPIO_NUM;
  config.pin_d1 = Y3_GPIO_NUM;
  config.pin_d2 = Y4_GPIO_NUM;
  config.pin_d3 = Y5_GPIO_NUM;
  config.pin_d4 = Y6_GPIO_NUM;
  config.pin_d5 = Y7_GPIO_NUM;
  config.pin_d6 = Y8_GPIO_NUM;
  config.pin_d7 = Y9_GPIO_NUM;
  config.pin_xclk = XCLK_GPIO_NUM;
  config.pin_pclk = PCLK_GPIO_NUM;
  config.pin_vsync = VSYNC_GPIO_NUM;
  config.pin_href = HREF_GPIO_NUM;
  config.pin_sscb_sda = SIOD_GPIO_NUM;
  config.pin_sscb_scl = SIOC_GPIO_NUM;
  config.pin_pwdn = PWDN_GPIO_NUM;
  config.pin_reset = RESET_GPIO_NUM;
  config.xclk_freq_hz = 20000000;
  config.pixel_format = PIXFORMAT_JPEG; 

  // Frame Size: VGA
  config.frame_size = FRAMESIZE_HVGA; 
  config.jpeg_quality = 12; 
  config.fb_count = 2; 

  // Init camera
  esp_err_t err = esp_camera_init(&config);
  if (err != ESP_OK) {
    Serial.printf("Camera init failed with error 0x%x", err);
    return;
  }

  // Start server routes
  server.on("/", handleRoot);
  server.on("/joystick1", handleJoystick1);
  server.on("/joystick2", handleJoystick2);
  server.on("/stream", handleStream);
  server.begin();
}

void loop() {
  server.handleClient(); // Handle incoming client requests
}
